Arduinoを使ってみる

複雑なことも簡単なものの組み合わせや応用だと改めて考えさせられる授業だった。
LEDを明るいときは消し、暗いときに光らせるコード
int ledPin = 13; // LEDをつなぐピン
int sensorPin = A1; // 光センサー（フォトレジスタ）
void setup() {
Serial.begin(9600);
pinMode(ledPin, OUTPUT);
}
void loop() {
int val = analogRead(sensorPin); // 明るさ取得
Serial.println(val); // しきい値（ここは調整する）
if (val > 300) { // 暗いとき
digitalWrite(ledPin, HIGH); // LED ON
} else { // 明るいとき
digitalWrite(ledPin, LOW); // LED OFF
}

delay(500);
}
